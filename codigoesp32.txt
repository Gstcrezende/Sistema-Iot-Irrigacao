#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <PubSubClient.h>
#include "DHT.h"

#define DHTPIN 15
#define DHTTYPE DHT22

DHT dht(DHTPIN, DHTTYPE);

const char* ssid = "Wokwi-GUEST";
const char* password = "";

const char* mqtt_server = "671be66b88cf41909db655fee73234dd.s1.eu.hivemq.cloud";
const int mqtt_port = 8883;
const char* mqtt_user = "iotuser";
const char* mqtt_pass = "Iot12345";

const char* topic_pub = "/fazenda/dados";
const char* topic_sub = "/fazenda/irrigacao";

WiFiClientSecure espClient;
PubSubClient client(espClient);

int sensorSolo = 34;
int rele = 2;

void callback(char* topic, byte* payload, unsigned int length) {
  String msg;
  for (int i = 0; i < length; i++) msg += (char)payload[i];

  if (msg == "ON") digitalWrite(rele, HIGH);
  if (msg == "OFF") digitalWrite(rele, LOW);
}

void reconnect() {
  while (!client.connected()) {
    if (client.connect("ESP32", mqtt_user, mqtt_pass)) {
      client.subscribe(topic_sub);
    } else {
      delay(2000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(rele, OUTPUT);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);

  espClient.setInsecure();

  client.setServer(mqtt_server, mqtt_port);
  client.setCallback(callback);

  dht.begin();
}

void loop() {
  if (!client.connected()) reconnect();
  client.loop();

  int solo = analogRead(sensorSolo);
  float umidadeSolo = map(solo, 0, 4095, 0, 100);

  float temp = dht.readTemperature();
  float umidadeAr = dht.readHumidity();

  String payload = "{";
  payload += "\"solo\":" + String(umidadeSolo) + ",";
  payload += "\"temp\":" + String(temp) + ",";
  payload += "\"ar\":" + String(umidadeAr);
  payload += "}";

  client.publish(topic_pub, payload.c_str());

  delay(3000);
}
